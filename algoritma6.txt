PERHITUNGAN DISKON BELANJA

DECLARE jumlahPembelian, totalPembelian, totalDiskon: num
DECLARE diskon:float
BEGIN
   	INPUT(jumlahPembelian)

	IF jumlahPembelian < 500000
		diskon = 0/100
		
	ELSEIF jumlahPembelian <= 500000 && jumlahPembelian >= 1000000
		diskon = 5/100

	ELSE
		diskon = 10/100

	ENDIF

	totalDiskon <- jumlahPembelian * diskon
	totalPembelian <- jumlahPembelian - totalDiskon	

	OUTPUT("DISKON = ", diskon)
	OUTPUT("TOTAL PEMBELIAN = ", totalPembelian)
END
